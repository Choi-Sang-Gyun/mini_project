CREATE TABLE USERS (
    USER_ID NUMBER NOT NULL PRIMARY KEY,
    EMAIL VARCHAR2(30),
    PASSWORD VARCHAR2(30),
    GENDER VARCHAR2(10),
    NICKNAME VARCHAR2(50),
    PROFILE_IMAGE BLOB,
    AGE NUMBER 
);
CREATE TABLE POST(
    POST_ID NUMBER, 
    CONTENT VARCHAR2(4000), 
    USER_ID NUMBER, 
    THUMBNAIL BLOB
);
